CREATE TABLE fornecedores(
    id int NOT null PRIMARY KEY AUTO_INCREMENT,
    nome text,
    razao_social text,
    cnpj int,
    cpf int,
    email text,
    contato text,
    idUsuario int,
    FOREIGN KEY(idUsuario) REFERENCES usuario(id)
);
